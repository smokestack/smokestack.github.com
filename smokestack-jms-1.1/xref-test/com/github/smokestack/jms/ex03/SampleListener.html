46  package com.github.smokestack.jms.ex03;
47  
48  import javax.jms.Message;
49  import javax.jms.MessageListener;
50  import javax.jms.TextMessage;
51  import javax.jms.JMSException;
52  
53  /**
54   * Sample <code>MessageListener</code> implementation
55   *
56   * @author <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
57   * @version $Revision: 1.2 $ $Date: 2005/11/18 03:28:01 $
58   */
59  public class SampleListener implements MessageListener {
60  
61      public void onMessage(Message message) {
62          if (message instanceof TextMessage) {
63              TextMessage text = (TextMessage) message;
64              try {
65                  System.out.println("Received: " + text.getText());
66              } catch (JMSException exception) {
67                  System.err.println("Failed to get message text: " + exception);
68              }
69          }
70      }
71  }
